function keyPressed(){
if(keyCode === RIGHT_ARROW){
pageNumber = pageNumber + 1;
}
if(keyCode === LEFT_ARROW){
if(pageNumber >= 0){
pageNumber = pageNumber - 1;
}
}
return false;
}//end of keyPressed

function timeIt(){
//breeze's timer
if(timer > 0 && pageNumber == 15){
timer--;
}
if(timer == 0){
pageNumber++;
}
//willows timer
if(timerW > 0 && pageNumber==19){

}

}

}

}

if(timerM == 0){

}
//Bears Timer

}

}
}
